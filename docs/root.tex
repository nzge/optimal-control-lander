\begin{abstract}
This project studies the optimal guidance and control of a planar variable-mass rocket/lander with translational and rotational dynamics, bounded controls, nonlinear coupling, free-final-time optimal control, endpoint manifold constraints, and trajectory tracking. The project combines Pontryagin’s Maximum Principle (PMP), controllability analysis, Linear Quadratic Regulation (LQR), minimum-time optimal control, nonlinear indirect optimal control, and numerical two-point boundary value problem (TPBVP) solvers. The project is inspired by powered-descent guidance and reusable rocket landing systems studied in modern aerospace optimal control literature. All theoretical derivations must be based on PMP. Numerical implementations may use indirect shooting, multiple shooting, continuation or homotopy methods, and indirect/direct hybrid approaches, provided that the optimality conditions are derived from PMP.
\end{abstract}

\section{Numerical Methodology}

\subsection{Linear Systems Solver: The Sweep Method (Parts I \& II)}
$scipy.integrate.solve\_ivp$ integrates the Riccati equation backward in time to generate the feedback gain matrix $K(t)$. An interpolator was then used on this gain during the forward-pass simulation of the state trajectory.

\subsection{Nonlinear TPBVP Solvers (Parts III \& IV)}
The Riccati substitution fails for nonlinear dynamics. Indirect Shooting was used to tackle the coupled state/costate equations.

\subsection{Constraint Enforcement Strategies}

\begin{itemize}
    \item Control Constraints ($T, \tau$): For the linear case, I applied post-calculation saturation (clipping the LQR commanded thrust). For nonlinear phases, PMP mathematically handles the bounds or how you passed them as boundary arrays to a collocation solver.
    \item State Constraints ($p_z \ge 0$, $m \ge m_{dry}$): Adjoining state constraints directly to the PMP Hamiltonian creates severe analytical difficulties (jump discontinuities in the costates). The numerical workaround is to either:
    \begin{itemize}
        \item "Soft Constraints" (adding massive penalties to your $Q$ matrix if the rocket goes underground)
        \item "Event Tracking" (terminating the ODE solver upon impact)
        \item Direct constraint arrays in an NLP solver.
    \end{itemize}
\end{itemize}

\subsection{Analytical Derivations (PMP)}

\begin{itemize}
    \item Hamiltonian construction
    \[H(x, u, p) = L(x, u) + p^\top f(x, u)\] 
    \vspace{2pt}\centerline{where $L(x,u)=x^\top Qx + u^\top Ru$}\vspace{2pt}
    \[H(x, u, p) = x^\top Q x + u^\top R u + p^\top (Ax + Bu)\]
    
    \item Transversality
    \[p(t_f) = \frac{\partial \phi}{\partial x}\bigg|_{t=t_f}\]
    \[p(t_f) = 2 Q_f x(t_f) \]
    
    \item Adjoint Equations (costate dynamics)
    \[\dot{p} = -\frac{\partial H}{\partial x}\]
    Using matrix calculus (and the symmetry of $Q$):
    \[\dot{p} = -\big( 2 Q x + A^\top p \big) = -2 Q x - A^\top p \]
    
    \item Stationarity Conditions
    \[\frac{\partial H}{\partial u} = 0 \implies 2 R u + B^\top p = 0 \implies u^* = -\frac{1}{2} R^{-1} B^\top p \]
    
    \item Riccati Differential equation
    Assumption (Riccati Substitution): 
    \[p(t) = P(t)x(t)\] 
    Where $P(t)$ is an unknown, time-varying symmetric matrix. Differentiate the assumption using the product rule:
    \[\dot{p} = \dot{P}x + P\dot{x}\]
    Substitute $\dot{p}$ and $\dot{x}$ (the dynamics).
    \[-2Qx - A^\top P x = \dot{P}x + P(Ax + Bu^*)\]
    Substitute the optimal control $u^*$. Remembering that $p = Px$, it becomes $u^* = -\frac{1}{2} R^{-1} B^\top P x$.
    \[-2Qx - A^\top P x = \dot{P}x + P\left(Ax - \frac{1}{2} B R^{-1} B^\top P x\right)\]
    \[-2Qx - A^\top P x = \dot{P}x + PAx - \frac{1}{2} P B R^{-1} B^\top P x\]
    Group all terms multiplying $x(t)$. Move everything to one side of the equation:
    \[\left[ \dot{P} + P A + A^\top P - \frac{1}{2} P B R^{-1} B^\top P + 2Q \right] x = 0\]
    Because this equation must hold true for any arbitrary state trajectory $x(t)$, the bracketed matrix expression must identically equal zero. Setting the bracket to zero yields the Riccati Differential Equation (RDE): 
    \[\boxed{\dot{P} = -P A - A^\top P + \frac{1}{2} P B R^{-1} B^\top P - 2Q}\]
    
    \item Optimal State-feedback law
    \[P(t_f) = 2 Q_f, \quad u^*(t) = -K(t) x(t), \quad K(t) = \frac{1}{2} R^{-1} B^\top P(t)\]
\end{itemize}

\subsection{Analysis}

Controllability Analysis

\begin{itemize}
    \item Algebraic Controllability Matrix (Frozen-Time):
    Used to check the structural control authority of the LTI system (or a "frozen" snapshot of an LTV system). For a system with $n = 7$ states:
    \[C = \begin{bmatrix} B & AB & A^2B & A^3B & A^4B & A^5B & A^6B \end{bmatrix}\]
    
    \item Finite-Horizon Controllability Gramian:
    \[W_c(0, t_f) = \int_{0}^{t_f} \Phi(0, \tau) B(\tau) B(\tau)^\top \Phi(0, \tau)^\top d\tau\]
    Where $\Phi(t, \tau)$ is the state transition matrix of $A(t)$.

    \item Numerical Computation (Differential Equation):
    To practically compute $W_c$ numerically and avoid explicitly calculating the integral and state transition matrix, we can solve its differential equation form:
    \[\dot{W}_c(t) = A(t)W_c(t) + W_c(t)A(t)^\top + B(t)B(t)^\top\]
    
    \item Minimum Control Energy:
    The minimum control energy $E$ required to reach a specific state deviation $x_f$ is given by:
    \[E = x_f^\top W_c^{-1} x_f\]
\end{itemize}

\section{PART II - Optimal Trajectory Tracking}

\subsection{Phase formulation}

Objective:
\[
    \begin{split}
    J = {}&\int_{0}^{t_f} \Big[ (x - x_{\text{ref}})^\top Q (x - x_{\text{ref}}) + u^\top R u \Big] dt \\
    &+ (x(t_f) - x_{\text{ref}}(t_f))^\top Q_f (x(t_f) - x_{\text{ref}}(t_f))
    \end{split}
\]

\subsection{Analytical Derivations (PMP)}

\begin{itemize}

    \item The Hamiltonian:
    \[H = (x - x_{\text{ref}})^\top Q (x - x_{\text{ref}}) + u^\top R u + p^\top (Ax + Bu)\]
    
    \item Adjoint Equation (Costate Dynamics):
    \[\dot{p} = -\frac{\partial H}{\partial x}\] 
    \[\dot{p} = -2Q(x - x_{\text{ref}}) - A^\top p \]
    
    \item Stationarity (Optimal Control Law):
    \[\frac{\partial H}{\partial u} = 0 \implies 2Ru + B^\top p = 0 \implies u^* = -\frac{1}{2} R^{-1} B^\top p\]
    
    \item Transversality (Terminal Condition):
    \[p(t_f) = \frac{\partial \phi}{\partial x} \bigg|_{t=t_f}\]
    \[p(t_f) = 2Q_f(x(t_f) - x_{\text{ref}}(t_f)) \]
    
    \item Tracking Riccati Equations (Riccati substitution)
    
        The added auxiliary vector $s(t)$ accounts for the tracking offset:
        \[p(t) = P(t)x(t) + s(t)\]
        \[\dot{p} = \dot{P}x + P\dot{x} + \dot{s}\]
    
        Substitute $\dot{p}$ and $\dot{x}$:
        \[-2Q(x - x_{\text{ref}}) - A^\top(Px + s) = \dot{P}x + P(Ax + Bu^*) + \dot{s}\]
    
        Substitute $u$ ($u^* = -\frac{1}{2}R^{-1}B^\top(Px + s)$):
        \[
            \begin{aligned}
            -2Qx &+ 2Qx_{\text{ref}} - A^\top P x - A^\top s ={} \\
            &\dot{P}x + PAx - \tfrac{1}{2}P B R^{-1} B^\top (Px + s) + \dot{s}
            \end{aligned}
        \]
        
        Group the terms by $x(t)$:
        \[
            \begin{split}
            &\Big[ \dot{P} + PA + A^\top P - \tfrac{1}{2}PBR^{-1}B^\top P + 2Q \Big] x(t) \\
            &\quad + \Big[ \dot{s} + A^\top s - \tfrac{1}{2}PBR^{-1}B^\top s - 2Qx_{\text{ref}} \Big] = 0
            \end{split}
        \]
    
        Since the expression must hold true for any arbitrary trajectory $x(t)$, we get the following tracking Riccati pair:
    
        \begin{itemize}
            \item  Matrix Riccati Differential Equation (The Feedback Rulebook):
            \[\dot{P} = -PA - A^\top P + \frac{1}{2}PBR^{-1}B^\top P - 2Q\]
            
            \item Auxiliary Vector Differential Equation (The Feedforward Tracker):
            \[\dot{s} = - \left( A^\top - \frac{1}{2}PBR^{-1}B^\top \right) s + 2Q x_{\text{ref}}\]
        \end{itemize}
        
    
    \item Closed-Loop Tracking Control Law

        Once you integrate both $\dot{P}$ and $\dot{s}$ backward from $t_f$ to $0$, you can construct your final control law. Substitute $p(t) = P(t)x(t) + s(t)$ back into $u$: 
        \[u^*(t) = -\frac{1}{2} R^{-1} B^\top P(t) x(t) - \frac{1}{2} R^{-1} B^\top s(t)\]
        This is elegantly split into two components:
        \[u^*(t) = -K(t)x(t) + u_{ff}(t)\]
        $K(t)x(t)$ is the feedback component that stabilizes the rocket and kills deviations. $u_{ff}(t)$ is the feedforward component that anticipates the control effort.
    
\end{itemize}

\subsection{Phase B — Return-and-Land Problem}

Phase B begins precisely where Phase A ends, thus the initial condition for the landing phase is $x(0) = x_1$.

\subsubsection*{Formulation \& Core PMP} 
\[J_B = \int_{0}^{t_f} \Big( x^\top Q x + u^\top R u \Big) dt\]

Generic Hamiltonian, and the fundamental $H(t_f) = 0$ condition:
\[H_B(x, u, p) = x^\top Q x + u^\top R u + p^\top (Ax + Bu)\]

The unconstrained control law and adjoint equations remain identical to Phase I:
\[u^*(t) = -\frac{1}{2} R^{-1} B^\top p(t)\]
\[\dot{p}(t) = -2Qx(t) - A^\top p(t)\]

Because $t_f$ is free, PMP imposes the free-time transversality condition. The Hamiltonian evaluated at the terminal time must equal zero: 

\[
\begin{aligned}
H_B(t_f) = 0 \implies & x(t_f)^\top Q x(t_f) + u(t_f)^\top R u(t_f) \\
& + p(t_f)^\top \big( Ax(t_f) + Bu(t_f) \big) = 0
\end{aligned}
\]

\subsubsection*{Transversality} 

\[p(t_f) = \frac{\partial \phi}{\partial x}\bigg|_{t_f} + \sum_{j} \nu_j \nabla h_j(x(t_f))\]

\begin{itemize}
    \item \textit{Manifold 1 (Pinpoint Landing)}

    \begin{itemize}
        \item Description
        \[
        \begin{aligned}
        h_1(x) &= p_z = 0 \\
        h_2(x) &= v_z = 0 \\
        h_3(x) &= \theta = 0
        \end{aligned}
        \]

        \item Gradients
        \[
        \begin{aligned}
        \nabla h_1 &= [0, 1, 0, 0, 0, 0, 0]^\top \\

        \nabla h_3 &= [0, 0, 0, 0, 1, 0, 0]^\top
        \end{aligned}
        \]

        \item Transversality
        \[
        \begin{aligned}
        p_{p_x}(t_f) &= \frac{\partial \phi}{\partial p_x} && \text{(Free state)} \\
        p_{p_z}(t_f) &= \frac{\partial \phi}{\partial p_z} + \nu_1 && \text{(Constrained)} \\
        p_{v_x}(t_f) &= \frac{\partial \phi}{\partial v_x} && \text{(Free state)} \\
        p_{v_z}(t_f) &= \frac{\partial \phi}{\partial v_z} + \nu_2 && \text{(Constrained)} \\
        p_{\theta}(t_f) &= \frac{\partial \phi}{\partial \theta} + \nu_3 && \text{(Constrained)} \\
        p_{\omega}(t_f) &= \frac{\partial \phi}{\partial \omega} && \text{(Free state)} \\
        p_{m}(t_f) &= \frac{\partial \phi}{\partial m} && \text{(Free state)}
        \end{aligned}

    \item \textit{Manifold 2 (Surface Landing)}

        \begin{itemize}
            \item Description: landing on the surface of a dome/sphere.

            h_1(x) &= (p_x - p_c)^2 + p_z^2 - r^2 = 0 \\
            h_2(x) &= v_z = 0 \\
            h_3(x) &= \theta = 0
            \end{aligned}
            \]

            \item The Gradients: $h_1$ couples $p_x$ and $p_z$, spanning 2 dimensions.
            \[
            \begin{aligned}
            \nabla h_1 &= [2(p_x - p_c), 2p_z, 0, 0, 0, 0, 0]^\top \\
            \nabla h_2 &= [0, 0, 0, 1, 0, 0, 0]^\top \\
            \nabla h_3 &= [0, 0, 0, 0, 1, 0, 0]^\top
            \end{aligned}
            \]

            \item Transversality
            \[
            \begin{aligned}
            p_{p_x}(t_f) &= \frac{\partial \phi}{\partial p_x} + 2\nu_1(p_x(t_f) - p_c) \\
            p_{p_z}(t_f) &= \frac{\partial \phi}{\partial p_z} + 2\nu_1 p_z(t_f)
            \end{aligned}

            \textit{(The remaining 5 costates are identical to M1)}

            \item Conclusion for M2: We can eliminate the unknown multiplier $\nu_1$ by dividing the first two equations (assuming $\frac{\partial \phi}{\partial x} = 0$). This yields the exact boundary condition passed to the TPBVP solver:
            \[\frac{p_{p_x}(t_f)}{p_{p_z}(t_f)} = \frac{p_x(t_f) - p_c}{p_z(t_f)}\]
        \end{itemize}  
\end{itemize}

\section{PART IV - The Nonlinear Optimal Control Near Hover}

\subsection{Phase Formulation}

Consider the fixed-time, free-endpoint nonlinear optimal control problem:

\[\min_{u} \int_{0}^{t_f} \left( x^\top Q x + u^\top R u \right) dt\]

Recall: 
State Equations (which is also $\dot{x} = \frac{\partial H}{\partial p}$):

\[
\dot{x} = \begin{bmatrix} 
v_x \\ 
v_z \\ 
-\frac{T}{m}\sin\theta \\
\frac{T}{m}\cos\theta - g \\ 
\omega \\ 
\frac{\tau}{I} \\ 
-\alpha T 
\end{bmatrix}
\]

System is anchored by the known initial launch state at $t=0$: 
\[x(0) = x_{0}\]
    
\subsection{Analytical Derivations}

\begin{itemize}
    \item Nonlinear Hamiltonian
    
    Adjoin the nonlinear dynamics to the running cost using the costate vector $p = [p_1, p_2, p_3, p_4, p_5, p_6, p_7]^\top$:

    \[
    \begin{aligned}
        H(x, u, p) ={}& x^\top Q x + u^\top R u + p_1 v_x + p_2 v_z - p_3 \tfrac{T}{m}\sin\theta \\
        &+ p_4 \big( \tfrac{T}{m}\cos\theta - g \big) + p_5 \omega + p_6 \tfrac{\tau}{I} - p_7 \alpha T
    \end{aligned}
    \]

    \item Adjoint equations (costate equations)
    \[\dot{p} = -\frac{\partial H}{\partial x}\]
    
    \[
    \begin{aligned}
    \dot{p}_1 &= -2Q_{11} p_x \\
    \dot{p}_2 &= -2Q_{22} p_z \\
    \dot{p}_3 &= -2Q_{33} v_x - p_1 \\
    \dot{p}_4 &= -2Q_{44} v_z - p_2 \\
    \dot{p}_5 &= -2Q_{55} \theta + p_3 \frac{T}{m}\cos\theta + p_4 \frac{T}{m}\sin\theta \\
    \dot{p}_6 &= -2Q_{66} \omega - p_5 \\
    \dot{p}_7 &= -2Q_{77} m - p_3 \frac{T}{m^2}\sin\theta + p_4 \frac{T}{m^2}\cos\theta
    \end{aligned}
    \]
    
    \item Stationarity conditions ($\frac{\partial H}{\partial u} = 0$)

    \begin{itemize}
        \item For Thrust ($T$):
        \[\frac{\partial H}{\partial T} = 2R_{11}T - p_3 \frac{\sin\theta}{m} + p_4 \frac{\cos\theta}{m} - p_7 \alpha = 0\]
        \[T^* = \text{sat}_{T_{\min}}^{T_{\max}} \left[ \frac{1}{2 R_{11}} \left( p_3 \frac{\sin\theta}{m} - p_4 \frac{\cos\theta}{m} + p_7 \alpha \right) \right]\]

        \item For Torque ($\tau$):
        \[\frac{\partial H}{\partial \tau} = 2R_{22}\tau + \frac{p_6}{I} = 0\]
        \[\tau^* = \text{sat}_{\tau_{\min}}^{\tau_{\max}} \left[ -\frac{p_6}{2 I R_{22}} \right]\]
    \end{itemize}

    \item Transversality (terminal condition)
    
    Because the terminal state is subject to optimization, the terminal costate at $t=t_f$ is governed by the transversality condition based on the terminal penalty $\phi(x(t_f))$ (or specific manifold constraints):
    \[p(t_f) = \frac{\partial \phi}{\partial x} \bigg|_{t=t_f}\]

    \item Complete TPBVP
    
    The complete TPBVP is formed by substituting the optimal, saturated control laws ($T^*$, $\tau^*$) back into the 7 state equations and 7 adjoint equations. This creates a deeply coupled system of 14 nonlinear, first-order ordinary differential equations. This system must be solved simultaneously across the domain $t \in [0, t_f]$, subject to the split boundary conditions $x(0)$ and $p(t_f)$.

    \begin{equation} \label{eq:full_tpbvp}
    \small
    \begin{aligned}
        % Forward State Dynamics
        \dot{p}_x &= v_x \\
        \dot{p}_z &= v_z \\
        \dot{v}_x &= -\frac{T}{m}\sin\theta \\
        \dot{v}_z &= \frac{T}{m}\cos\theta - g \\
        \dot{\theta} &= \omega \\
        \dot{\omega} &= \frac{\tau}{I} \\
        \dot{m} &= -\alpha T \\[1.5ex] 
        % Backward Costate Dynamics
        \dot{p}_1 &= -2Q_{11} p_x \\
        \dot{p}_2 &= -2Q_{22} p_z \\
        \dot{p}_3 &= -2Q_{33} v_x - p_1 \\
        \dot{p}_4 &= -2Q_{44} v_z - p_2 \\
        \dot{p}_5 &= -2Q_{55} \theta + \frac{T}{m} \big(p_3 \cos\theta + p_4 \sin\theta \big) \\
        \dot{p}_6 &= -2Q_{66} \omega - p_5 \\
        \dot{p}_7 &= -2Q_{77} m + \frac{T}{m^2} \big(p_4 \cos\theta - p_3 \sin\theta \big)
    \end{aligned}
    \end{equation}
    
\end{itemize}
